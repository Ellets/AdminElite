// PASTE YOUR GOOGLE DEPLOYMENT URL HERE const SCRIPT_URL = "YOUR_NEW_DEPLOY_URL_HERE"; async function doUpload() { const fileInput = document.getElementById('fileInput'); const file = fileInput.files[0]; if (!file) return alert("Please select a file."); const status = document.getElementById('status'); status.innerText = "Step 1: Uploading... please wait."; const reader = new FileReader(); reader.onload = async function(e) { const payload = { action: 'upload', base64: e.target.result.split(',')[1], mimeType: file.type, fileName: file.name }; try { const res = await fetch(SCRIPT_URL, { method: "POST", body: JSON.stringify(payload) }); const result = await res.json(); if(result.status === "success") { status.innerText = "Upload Complete! Ready to Fetch."; document.getElementById('nameInput').value = file.name; } } catch (err) { status.innerText = "Upload Error. Check Console."; console.error(err); } }; reader.readAsDataURL(file); } async function doFetch() { const name = document.getElementById('nameInput').value; const status = document.getElementById('status'); if (!name) return alert("Enter a filename to fetch."); status.innerText = "Step 2: Fetching link..."; try { const res = await fetch(SCRIPT_URL, { method: "POST", body: JSON.stringify({ action: 'fetch', fileName: name }) }); const result = await res.json(); if(result.status === "success") { status.innerText = "Link Found!"; document.getElementById('outputLink').value = result.url; } else { status.innerText = "File not found in Drive."; } } catch (err) { status.innerText = "Fetch Error."; console.error(err); } }
